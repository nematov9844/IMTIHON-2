import { JSDOM } from 'jsdom'; import { fireEvent, getByText } from '@testing-library/dom'; const dom = new JSDOM(` ${document.body.innerHTML} `); global.document = dom.window.document; global.window = dom.window; describe('Header', () => { test('toggles dropdown on click', () => { const dropdownBtn = getByText(document.body, 'UZ'); fireEvent.click(dropdownBtn); expect(document.querySelector('.ru').classList).not.toContain('none'); expect(document.querySelector('.emg').classList).not.toContain('none'); }); test('toggles mobile menu on click', () => { const toggleBtn = getByText(document.body, /bars/i); fireEvent.click(toggleBtn); expect(document.querySelector('.header__list').classList).toContain('show'); }); }); describe('Hero section', () => { test('renders title and description', () => { const title = getByText(document.body, /xalqaro darajagacha/i); const description = getByText(document.body, /O'zbekistonda qizlarni akt sohasiga/i); expect(title).toBeInTheDocument(); expect(description).toBeInTheDocument(); }); }); describe('Alliance section', () => { test('renders cards with correct content', () => { const cards = document.querySelectorAll('.alliance__card'); expect(cards.length).toBeGreaterThan(0); cards.forEach((card) => { expect(card.textContent).not.toBe(''); }); }); }); describe('Virtual section', () => { test('renders join and create profile buttons', () => { const joinBtn = getByText(document.body, /A'zo bo'lish/i); const profileBtn = getByText(document.body, /Profile yaratish/i); expect(joinBtn).toBeInTheDocument(); expect(profileBtn).toBeInTheDocument(); }); }); describe('Footer', () => { test('renders logo and links', () => { const logo = getByText(document.body, /It ta'lim uyushmasi/i); const links = document.querySelectorAll('.footer__link'); expect(logo).toBeInTheDocument(); expect(links.length).toBeGreaterThan(0); }); });
